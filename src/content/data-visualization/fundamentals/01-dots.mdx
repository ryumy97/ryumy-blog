# 1차원 데이터 표현 - 점

1차원 데이터를 사용하여 가장 기초적인 시각화를 해보자.

## 1차원 데이터의 기본 개념

1차원 데이터란 값이 하나인 데이터를 말합니다.

예를 들어, 만약 우리가 버튼이 눌렸나 안눌렸나를 확인하는 것을 컴퓨터가 알아들을수 있는 데이터로 기져온다면, Boolean 타입으로 저장할수 있습니다.
혹은 오늘 하루는 몇칼로리 먹었느냐를 저장한다면 Number 타입으로 저장할수 있습니다.

이런 데이터들은 모두 1차원 데이터라고 우리는 정의할수 있습니다. 하나의 데이터가 얼마만큼 있느냐 (혹은 있거나 없거나) 라는 개념인것이죠.

## 점

점은 1차원 데이터를 표현하는 가장 기초적인 요소입니다.

있다와 없다를 표현할수 있는 요소입니다.

예) 버튼이 눌렸나 안눌렸나를 표현하는 것을 컴퓨터가 알아들을수 있는 데이터로 기져온다면, Boolean 타입으로 저장할수 있습니다.

## 조금 더 범용적인 표현방식

우리는 어떻게 해야 데이터 표현을 더 범용적으로 할수 있을까요?
가장 쉬운방법은 일단 데이터와 표현을 분리해야 합니다.
데이터를 저장하는 변수와 그 데이터를 변화시키는 코드를 함수화 하고,
데이터를 표현하는 코드를 함수화 하는 것입니다.

![/01-dots/figure-01.png](/images/data-visualization/fundamentals/01-dots/figure-01.png "점. 표기-01) 데이터와 표현을 분리.")

점-2를 리팩토링 하여 데이터와 표현을 분리해보겠습니다.

  `}

      display: flex;
      flex-direction: column;
      align-items: center;
      gap: 20px;
    }
    .counter-container {

    }
    button {

    }

  js={`
    import "./styles.css";

    const canvas = document.getElementById('dot-3');
    const ctx = canvas.getContext('2d');

    const button = document.getElementById('button');
    const reset = document.getElementById('reset');

    canvas.width = 200;
    canvas.height = 200;

    ctx.fillStyle = 'black';

    // 버튼이 눌린 횟수를 저장하는 변수
    let count = 0;

    // 버튼이 눌린 횟수를 늘린다.
    function increaseCount() {
      count++;
    }

    // 버튼이 눌린 횟수를 초기화 한다.
    function resetCount() {
      count = 0;
    }

    // 캔버스를 초기화한다.
    function initCanvas() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      ctx.beginPath();
      ctx.moveTo(0, 180);
      ctx.lineTo(canvas.width, 180);
      ctx.strokeStyle = '#ddd';
      ctx.stroke();
    }

    // count 변수에 따라 다음 점을 그린다.
    function drawDot(count) {
      ctx.fillStyle = 'steelblue';
      ctx.beginPath();
      ctx.ellipse(100, 180 - count * 10, 5, 5, 0, 0, 2 * Math.PI);
      ctx.closePath();
      ctx.fill();
    }

    // count 변수를 텍스트로 표시한다.
    function updateCounterText(count) {
      document.getElementById('counter').textContent = count.toString();
    }

    // 캔버스 초기화
    initCanvas();

    // 버튼이 눌렸을 때
    button.addEventListener('click', () => {
      // count 변수를 증가시킨다.
      increaseCount();

      // count 변수에 따라 점을 그린다.
      drawDot(count);

      // count 변수를 텍스트로 표시한다.
      updateCounterText(count);
    });

    // 초기화
    reset.addEventListener('click', () => {
      // count 변수를 초기화 한다.
      resetCount();

      // 캔버스를 초기화한다.
      initCanvas();

      // count 변수를 텍스트로 표시한다.
      updateCounterText(count);
    });

`}
/>

예제 점-3에서 라인 14에서 부터 24까지를 보면, count 변수를 선언하고, 초기화 하는 함수와 증가시키는 함수를 정의하고 있습니다.

라인 27에서 부터 36까지를 보면, 캔버스를 초기화하고 기본 선을 그리는 함수를 정의하고 있습니다.

라인 38 함수 drawDot 는 count 변수를 캔버스에 표현하는 함수입니다.

라인 47 함수 updateCounterText 는 count 변수를 html 텍스트로 표시하는 함수입니다.

만약에 데이터가 1 부터 10까지 무작위하게 생성되었다면 어떻게 해야 할까요?
앞서서 데이터와 변수를 분리를 했으니, 우리는 쓴 데이터를 선언하고 제어하는 함수를 무작위 데이터를 생성하는 함수로 바꿔야 합니다.

  title="점 - 4"
  height={460}
  html={`

            Random Number: <span id="random-number">0</span>

  `}
  css={`
    .container {
      padding: 20px;

      flex-direction: column;
      align-items: center;
      gap: 20px;
    }
    .number-container {

    }
    canvas {
      border: 1px solid #ccc;
    }
    button {
      padding: 10px 20px;
      background-color: #007bff;
      color: white;
      border: none;
      cursor: pointer;
    }
    .button-container {
      display: flex;
      flex-direction: row;
      align-items: center;
      gap: 10px;
    }
    #reset {
      background-color: #aaaaaa;
    }
  `}
  js={`
    import "./styles.css";

    const canvas = document.getElementById('dot-4');
    const ctx = canvas.getContext('2d');

    const button = document.getElementById('button');

    canvas.width = 200;
    canvas.height = 200;

    ctx.fillStyle = 'black';

    // 무작위 데이터를 저장하는 변수
    let randomNumber = 0;

    // 무작위 데이터를 생성하는 함수
    function generateRandomNumber() {
      randomNumber =  Math.floor(Math.random() * 10) + 1;
    }

    // 무작위 데이터를 생성한다.
    generateRandomNumber();

    // 캔버스를 초기화한다.
    function initCanvas() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      ctx.beginPath();
      ctx.moveTo(0, 180);
      ctx.lineTo(canvas.width, 180);
      ctx.strokeStyle = '#ddd';
      ctx.stroke();
    }

    // randomNumber 변수에 따라 다음 점을 그린다.
    function drawDot(index) {
      ctx.fillStyle = 'steelblue';
      ctx.beginPath();
      ctx.ellipse(100, 180 - index * 10, 5, 5, 0, 0, 2 * Math.PI);
      ctx.closePath();
      ctx.fill();
    }

    // randomNumber 변수를 텍스트로 표시한다.
    function updateNumberText(num) {
      document.getElementById('random-number').textContent = num.toString();
    }

    // randomNumber 변수를 표현하는 함수 묶음
    function render() {
      initCanvas();

      for (let i = 0; i < randomNumber; i++) {
        drawDot(i);
      }

      updateNumberText(randomNumber);
    }

    // 캔버스 초기화
    render();

    // 버튼이 눌렸을 때
    button.addEventListener('click', () => {
      // 무작위 데이터를 생성한다.
      generateRandomNumber();

      // 새로운 데이터를 표현한다
      render();
    });

`}
/>

예제 점-4의 바뀐점이라면 크게 두가지가 되는데
첫번째는 위에서 말한대로 데이터를 0~10까지의 무작위 데이터로 바꾼것이고,
두번째는 표현하는 함수를 묶어서 하나의 함수로 만든것입니다.

이렇게 하면 데이터가 무엇이던 상관없이 그것이 1차원 적인 데이터라면 우리는 점을 찍어서 표현을 할수있게 된것이죠

## 마무리

축하합니다! 이렇게 우리는 1차원 데이터를 표현하는 가장 기초적인 방법을 알아보면서,
범용성 있는 코드를 작성하는 방법을 알아보았습니다.

데이터와 표현을 분리해서, 데이터를 관리하는 코드를 작성하고, 데이터를 표현하는 코드를 작성하는 작동방식은
데이터 표현의 가장 핵심적인 부분입니다.
